\subsubsection{UCS 3 - Selezione dell'organizzazione}%kite level
\begin{itemize}
    \item \textbf{Attori primari:} Amministratore visualizzatore
    \item \textbf{Precondizione:} L'amministratore gestore si trova nella sezione di gestione delle organizzazioni\ap{G} e deve disporre di almeno un'organizzazione\ap{G}.
    \item \textbf{Postcondizione:} L'amministratore ha selezionato l'organizzazione\ap{G} desiderata.
    \item \textbf{Scenario principale:} L'amministratore deve scegliere l'organizzazione\ap{G} tra quelle disponibili.
    \item \textbf{Flusso di eventi:}
    \begin{enumerate}
        \item L'amministratore seleziona l'organizzazione\ap{G} desiderata tra quelle disponibili.
    \end{enumerate}
\end{itemize}
