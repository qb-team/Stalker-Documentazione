\subsubsection{UCS 3 - Selezione dell'organizzazione}%kite level
\begin{itemize}
    \item \textbf{Attori primari:} Amministratore visualizzatore
    \item \textbf{Precondizione:} L'amministratore visualizzatore si trova nella sezione di gestione delle \glo{organizzazioni} e deve disporre di almeno un'\glo{organizzazione}.
    \item \textbf{Postcondizione:} L'amministratore ha selezionato l'\glo{organizzazione} desiderata.
    \item \textbf{Scenario principale:} L'amministratore deve scegliere l'\glo{organizzazione} tra quelle disponibili.
    \item \textbf{Flusso di eventi:}
    \begin{enumerate}
        \item L'amministratore seleziona l'\glo{organizzazione} desiderata tra quelle disponibili.
    \end{enumerate}
\end{itemize}
